def swap_first_n_chars(s1, s2, n):
    # Ensure n is not greater than the length of either string
    if n > len(s1) or n > len(s2):
        raise ValueError("n is greater than the length of one or both strings")

    # Split and swap
    new_s1 = s2[:n] + s1[n:]
    new_s2 = s1[:n] + s2[n:]

    return new_s1, new_s2

# Example usage
string1 = "abcdef"
string2 = "uvwxyz"
n = 3

new_string1, new_string2 = swap_first_n_chars(string1, string2, n)
print("After swapping:")
print("String 1:", new_string1)
print("String 2:", new_string2)
